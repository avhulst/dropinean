CREATE TABLE IF NOT EXISTS `ean2id` (
  `id` int(11) NOT NULL AUTO_INCREMENT,
  `ean` varchar(64) NOT NULL,
  `productid` varchar(64) NOT NULL,
  `date` timestamp NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  PRIMARY KEY (`id`),
  UNIQUE KEY `ean` (`ean`)
) ENGINE=InnoDB  DEFAULT CHARSET=latin1;

CREATE TABLE  `dropinean`.`productdata` (
`id` INT NOT NULL AUTO_INCREMENT PRIMARY KEY ,
`sku` VARCHAR( 128 ) NOT NULL ,
`price` FLOAT NOT NULL ,
`uvp` FLOAT NOT NULL ,
`brand` VARCHAR( 128 ) NOT NULL ,
`name` VARCHAR( 128 ) NOT NULL ,
`backupsizerange` VARCHAR( 64 ) NOT NULL ,
`sizerange` VARCHAR( 64 ) NOT NULL ,
`size` VARCHAR( 64 ) NOT NULL ,
`lager` INT NOT NULL ,
`ident` INT NOT NULL ,
`dbinsert` TIMESTAMP ON UPDATE CURRENT_TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP ,
UNIQUE (
`ident`
)
) ENGINE = INNODB;

CREATE TABLE IF NOT EXISTS `ean` (
  `id` int(11) NOT NULL AUTO_INCREMENT,
  `ean` varchar(48) NOT NULL,
  `sku` varchar(72) NOT NULL,
  PRIMARY KEY (`id`),
  UNIQUE KEY `sku` (`sku`)
) ENGINE=InnoDB  DEFAULT CHARSET=latin1;
